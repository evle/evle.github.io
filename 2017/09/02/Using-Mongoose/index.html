Using Mongoose

Node.js and MongoDB are a pair made for each other，Mongoose allows us to have access to the MongoDB commands for CRUD simply and easily.

$ npm install mongoose --save    # install mongoose

Import Mongoose

const mongoose = require('mongoose')    
mongoose.connect('mongodb://localhost/mydatabase');

Defining a Model

Before we can handle CRUD operations, we will need a mongoose Model. These models are constructors that we define. They represent documents which can be saved and retrieved from our database.

Terms Note
Mongoose Schema used to define attributes for documents
Mongoose Methods can be defined on a mongoose schema

Models/user.js

var mongoose = require('mongoose')
var Schema = mongoose.Schema

var userSchema = new Schema({
  name: String,
  username: { type: String, required: true, unique: true },
  password: { type: String, required: true },
  created_at: Date,
  updated_at: Date
})

userSchema.methods.dudify = function() {
  this.name = this.name + '-dude'
  return this.name
};

userSchema.pre('save', function(next) {
var currentDate = new Date();

this.updated_at = currentDate;
if (!this.created_at)
  this.created_at = currentDate;

  next();
});

userSchema.index({username: 1})
module.exports = mongoose.model('User', userSchema)

Usage

var User = require('./app/models/user')

var evle = new User({
  name: 'evle',
  username: 'evlefsp',
  password: 'password'
})

evle.dudify(function(err, name) {
  if (err) throw err;
  console.log('Your new name is ' + name)
})

evle.save(function(err) {
  if (err) throw err;
  console.log('User saved successfully!')
})

Read

Find All

User.find({}, function(err, users) {
  if (err) throw err;
  console.log(users)
});

Find One

// find by user name
User.find({ username: 'evle' }, function(err, user) {
  if (err) throw err;
  console.log(user);
});

// find by id
User.findById(1, function(err, user) {
  if (err) throw err;
  console.log(user);
});

Using MongoDB query syntax.

example: get any admin that was created in the past month

// get the date 1 month ago
var monthAgo = new Date();
monthAgo.setMonth(monthAgo.getMonth() - 1);

User.find({ admin: true }).where('created_at').gt(monthAgo).exec(function(err, users) {
  if (err) throw err;

  // show the admins in the past month
  console.log(users);
});

Update

// get a user with ID of 1
User.findById(1, function(err, user) {
  if (err) throw err;

  // change the users location
  user.location = 'uk';

  // save the user
  user.save(function(err) {
    if (err) throw err;

    console.log('User successfully updated!');
  });

});

Find and Update

User.findOneAndUpdate({ username: 'evle' }, { username: 'evlefsp' }, function(err, user) {
  if (err) throw err;
  console.log(user);
});

// Find By ID and Update
User.findByIdAndUpdate(4, { username: 'evlefsp' }, function(err, user) {
  if (err) throw err;
  console.log(user);
});

Delete
